\section{Law of Harmony -- Үндүштөк мыйзамы}

\textbf{The vowels and consonants in a word influence each other and have the
same phonetic characteristics}

\begin{itemize}
    \item the vowel in the first syllable determines the remaining consonants
    \item if it begins with a hard vowel (а, ы, о, у) all the other vowels are
        hard
    \item if it starts with a soft vowel (э, е, и, ө, ү), the remaining ones
        are soft too
    \item loan words do not obey the law of harmony
    \item endings also follow the law of harmony
    \item the last vowel influences the vowel of the ending(s)
    \item the last consonant influences the kind of vowel that should be in the
        ending
    \item 2 types of vowels:
        \begin{itemize}
            \item round: а, о, э/е, ө
            \item square: ы, и, у, ү
        \end{itemize}
\end{itemize}

\section{Plural endings}

\begin{itemize}
    \item plural endings are not used for numbers or \textit{many, few...}
\end{itemize}

\section{Asking about someone's age -- Жашты суроо}

\subsection{Questions}

\begin{center}
\begin{tabular}{| l | l |}\hline
    Сен канча жаштасың?     & How old are you?                  \\\hline
    Сен канчадасың?         & -- (short)                        \\\hline
    Сиз канча жаштасыз?     & -- (formal)                       \\\hline
    Сиз канчадасыз?         & -- (formal, short)                \\\hline
    Ал канчажашта?          & How old is he/she/it?             \\\hline
    Ал канчада?             & -- (short)                        \\\hline
    Сенин атаң канчада?     & How old is your father?           \\\hline
    Сенин эжең канчада?     & How old is your older sister?     \\\hline
    Сиздин балаңыз канчада? & How old is your child?            \\\hline
\end{tabular}
\end{center}

\begin{itemize}
    \item to ask about someones something, add "ң" to the end of it
\end{itemize}

\subsection{Answers}

\begin{dialogue}
    \speak{A} Сен канчадасың?
    \speak{B} Менин жашым жыйырма бир.
\end{dialogue}

\begin{dialogue}
    \speak{A} Сен канчадасың?
    \speak{B} Менин жыйырма бир--демын.
\end{dialogue}

\begin{dialogue}
    \speak{A} Ал канчада?
    \speak{B} Ал <number>--да.
\end{dialogue}

\begin{dialogue}
    \speak{A} Сенин атаң канчада?
    \speak{B} Менин атам алтымыш алты--да.
\end{dialogue}

\begin{dialogue}
    \speak{A} Сенин апаң канчада?
    \speak{B} Менин апам алтымыш алты--да.
\end{dialogue}

\begin{dialogue}
    \speak{A} Сенин эжең канчада?
    \speak{B} Менин эжем отуз беш--те.
\end{dialogue}

\begin{dialogue}
    \speak{A} Сенин карындашың канчада?
    \speak{B} Менин карындашым он сегиз--де.
\end{dialogue}

\begin{itemize}
    \item to answer, add "--дамын" to the end of the number and pay attention
        to the law of harmony
    \item answering for other people you only attach "--да" to the number
    \item to answer about someone, attach "м" to their title
\end{itemize}

\section{Asking about birth years -- Жылын суроо}

\subsection{Questions}

\begin{center}
\begin{tabular}{| l | l |}\hline
    Сен канчанч жылы төрөлгөнсүң?   & In which year were you born?  \\\hline
    Сен канчанчы жылкысың?          & --(short)                     \\\hline
    Сиз канчанчы жылкысыз?          & --(short, formal)             \\\hline
\end{tabular}
\end{center}

\subsection{Answers}

\begin{center}
\begin{tabular}{| l | l |}\hline
    Мен эки миң биринчи жылы төрөлгөмүн.    & I was born in 2002.   \\\hline
    Мен эки миң биричи жылкымын.            & --(short)             \\\hline
    Мен токсон төртүнчү жылкымын            & I was born in 84.     \\\hline
\end{tabular}
\end{center}

\section{Asking about family status -- Үй бүлө абалын суроо}

\subsection{Questions}

\begin{center}
\begin{tabular}{| l | l |}\hline
    Үй бүлөң барбы?         & Do you have a family?         \\\hline
    Үй бүлөңүз барбы?       & --(formal)                    \\\hline
    Үйлөнгөнсүңбү?          & Are you married? (for men)    \\\hline
    Үйлөнгөнсүзбү?          & --(formal)                    \\\hline
    Турмушка чыккансыңбы?   & Are you married? (for women)  \\\hline
    Турмушка чыккансызбы?   & --(formal)                    \\\hline
    Балдарың барбы?         & Do you have children?         \\\hline
\end{tabular}
\end{center}

\subsection{Answers}

\begin{center}
\begin{tabular}{| l | l |}\hline
    Ооба, үй--бүлөм бар.        & Yes I have a family.      \\\hline
    Жок үй--бүлөм жок.          & No, I don't have a family.\\\hline
    Ооба, үйлөнгөмүн.           & Yes I am married.         \\\hline
    Жок үйлөнө элекмин.         & No I am not married.      \\\hline
    Ооба турмушка чыкканмын.    & Yes I am married.         \\\hline
    Жок турмушка чыга элекмин.  & No I am not married.      \\\hline
\end{tabular}
\end{center}

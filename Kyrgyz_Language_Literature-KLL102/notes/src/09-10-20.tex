\subsection{Example Conversation 1}

\begin{dialogue}
    \speak{Таалай} Саламатсызбы, чоң кыз? -- How are you, ma'am?
    \speak{Гулназ} Саламатчылык. -- Good, how are you?
    \speak{Таалай} Таанышып алалы. Менин атым Таалай. -- Let's get acquainted. My
    name is Taalai.
    \speak{Гулназ} Менин атым Гулназ. -- My name is Gulnaz.
    \speak{Таалай} Таанышканыма кубанычтамын. -- Pleasure to meet you.
    \speak{Гулназ} Мен дагы. -- Pleasure is mine.
\end{dialogue}

\subsection{Example Conversation 2}

\begin{dialogue}
    \speak{X}       Саламатсызбы? Сиздин атыңыз ким? -- How are you? What's
    your name?
    \speak{Бакыт}   Менин атым Бакыт. -- My name is Bakyt.
    \speak{X}       Кайталап коюңузчу. -- Please say that again.
    \speak{Бакыт}   Бакыт. -- Bakyt.
    \speak{X}       Рахмат. -- Thank you.
\end{dialogue}

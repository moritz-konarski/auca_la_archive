\section{Facts}

\begin{itemize}
    \item 36 letters
    \item 14 vowels
    \item 25 consonants
    \item whole Cyrillic alphabet + Ө, Ү, Ң
    \item two different sounds for K, Г, Ж
    \item 39 different sounds
    \item useful online keyboard:
        \url{https://www.lexilogos.com/keyboard/kyrgyz.htm}
\end{itemize}

\section{Alphabet}

The letters are mostly pronounced like in Russian.

\begin{itemize}
    \item Аа, Бб, Вв,
    \item Гг -- At the beginning of words, like "g" in "grin." Between vowels, 
        like a cross between "g" and the "h" in hat, or a sort of gurgle like 
        the French "r".
    \item Дд, Ее, Ёё
    \item Жж -- Like the "j" in "judge." (In loan words from Russian, like the 
        "ge" in "rouge.")
    \item Зз, Ии, Йй, Кк, Лл, Мм, Нн
    \item Ӊӊ -- Like the "ng" in "sang."
    \item Оо
    \item Өө -- Like the vowel in English "turn," or, like the German "ö".
    \item Пп, Рр, Сс, Тт, Уу
    \item Үү -- Like the German "ü".
    \item Фф, Хх, Цц, Чч, Шш, Щщ, Ъъ, Ыы, Ьь, Ээ, Юю, Яя
\end{itemize}

\section{Law of Harmony}

A word can only have either hard or soft vowels (\ref{hard_vowels}) in them.

\begin{itemize}
    \item Т\textbf{оо}л\textbf{о}р, \textbf{А}п\textbf{а}м\textbf{а}
    \item \textbf{Ү}йл\textbf{ө}рг\textbf{ө},
        К\textbf{и}т\textbf{е}пк\textbf{е}
    \item \textbf{А}й\textbf{ы}л-г\textbf{а}, \textbf{О}т\textbf{у}р\textbf{а}т
\end{itemize}

\section{Consonants}

\subsection{Consonant Б}

If Б stands between two vowels it sounds like the English 'W'.
\begin{center}
    \begin{tabular}{| l | l || l | l |}\hline
        Ооба -- оова & yes & Аба -- ава & air   \\\hline
        Обон -- овон & melody & & \\\hline
    \end{tabular}
\end{center}

At the beginning of a word is sounds like 'B'.
\begin{center}
    \begin{tabular}{| l | l || l | l |}\hline
        Бала & boy & Бар & have \\\hline
        Бүтүр & finish & Бер & give \\\hline
    \end{tabular}
\end{center}

If Б stands at the end of a word, it sounds like 'P'.
\begin{center}
    \begin{tabular}{| l | l || l | l |}\hline
        Клуб -- клуп & club & Араб -- Арап & Arabic \\\hline
    \end{tabular}
\end{center}

\subsection{Consonants Г/К}

If these consonants come before or after \textbf{hard} vowels (а, ы, о, у), 
they have a hard sound.
\begin{center}
    \begin{tabular}{| l | l || l | l |}\hline
        Ак & white & Кара & black   \\\hline
        Кыргыз & & Кол & hand       \\\hline
        Тогуз & nine & Кагаз & paper\\\hline
    \end{tabular}
\end{center}

If these consonants come before or after \textbf{soft} vowels (е, и, ө, ү), 
they have a soft sound.
\begin{center}
    \begin{tabular}{| l | l || l | l |}\hline
        Кел & гел & Кир & гир   \\\hline
        Көр & гөр & Күл & гүл   \\\hline
    \end{tabular}
\end{center}

\section{Singular and Plural Nouns}

If a noun ends in a vowel or in Й or Р, it ends in лар, лор, лер, or лөр
depending on the law of harmony.
\begin{center}
    \begin{tabular}{| l | l || l | l |}\hline
        Апа -- лар & mothers & Үй -- лөр & houses \\\hline
        Эже -- лер & sisters & Тоо -- лор & mountains \\\hline
    \end{tabular}
\end{center}

If a noun ends with a voiced consonants, it ends in дар, дор, дер, or дөр
depending of the law of harmony.
\begin{center}
    \begin{tabular}{| l | l || l | l |}\hline
        Кыз -- дар & girls & Стол -- дор & chairs \\\hline
        Гүл -- дөр & flowers & Калем -- дер & pencils \\\hline
    \end{tabular}
\end{center}

If a noun ends with an unvoiced consonant, it ends in тар, тор, тер, төр.
\begin{center}
    \begin{tabular}{| l | l || l | l |}\hline
        Китеп -- тер & books    & Сабак -- тар & lessons    \\\hline
        Сүрөт -- төр & pictures & дос -- тор & friends      \\\hline
    \end{tabular}
\end{center}

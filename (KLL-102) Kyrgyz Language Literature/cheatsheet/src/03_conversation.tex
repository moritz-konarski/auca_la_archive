\section{Conversation}

\subsection{Names}

\begin{itemize}
    \item we use Ким? as the question word
    \item asking about someones name
        \begin{table}[H]
            \center
        \begin{tabular}{| l | l |}\hline
            Kyrgyz              & English                   \\\hline\hline
            Сенин атын ким?     & What is your name?        \\\hline
            Анын аты ким?       & What is his/her name?     \\\hline
        \end{tabular}
        \caption{Asking about someones name}
        \end{table}
    \item telling ones name
        \begin{table}[H]
            \center
        \begin{tabular}{| l | l |}\hline
            Kyrgyz                  & English           \\\hline\hline
            Менин атым \ldots       & My name is \ldots \\\hline
            Сенин атын \ldots       & Your name is \ldots \\\hline
            Анын аты \ldots         & His/her name is \ldots \\\hline
        \end{tabular}
        \caption{Telling someones name}
        \end{table}
    \item example conversation
        \begin{dialogue}
            \speak{A} Саламатсызбы, чоң кыз?
            \speak{B} Саламатчылык.
            \speak{A} Таанышып алалы. Менин атым Таалай.
            \speak{B} Менин атым Гулназ.
            \speak{A} Таанышканыма кубанычтамын!
            \speak{B} Мен дагы.
        \end{dialogue}
    \item another example conversation
        \begin{dialogue}
            \speak{A} Саламатсызбы? Сиздин атыңыз ким?
            \speak{B} Менин атым Бакыт.
            \speak{A} Кайталап коюңузчу?
            \speak{B} Бакыт.
            \speak{A} Рахмат!
        \end{dialogue}
\end{itemize}

\subsubsection{Questions}

\begin{itemize}
    \item there are multiple ways one can ask about age
    \item one option is to simply as for the years (like in English)
        \begin{table}[H]
            \center
            \begin{tabular}{| l | l |}\hline
                Kyrgyz              & English       \\\hline\hline
                Сен канча жаштасың? & How old are you? (informal)   \\\hline
                Сен канчадасың?     & How old are you? (informal)   \\\hline
                Сиз канча жаштасыз? & How old are you? (formal)     \\\hline
                Сиз канчадасыз?     & How old are you? (formal)     \\\hline
                Ал канчажашта?      & How old is he / she / it?     \\\hline
                Ал канчада?         &  How old is he / she / it?    \\\hline
                Сенин атаң канчада? & How old is your father?       \\\hline
                Сенин эжең канчада? & How old is your older sister? \\\hline
                Сиздин балаңыз канчада? & How old is your child?    \\\hline
            \end{tabular}
            \caption{Phrases to ask about age}
        \end{table}
    \item the other option is to ask for the birth year
        \begin{dialogue}
            \speak{A} Сен канчанч жылы төрөлгөнсүң? --- In which year
            were you born?
            \speak{B} Мен эки миң биринчи жылы төрөлгөмүн. --- I was
            born in 2001.
        \end{dialogue}
        \begin{dialogue}
           \speak{A} Сен канчанчы жылкысың? --- In which year
           were you born?
           \speak{B} Мен эки миң биричи жылкымын. --- I was
           born in 2001.
        \end{dialogue}
        \begin{dialogue}
           \speak{A} Сиз канчанчы жылкысыз? --- In which year
           were you born?
           \speak{B} Мен токсон төртүнчү жылкымын. --- I was
           born in '94.
        \end{dialogue}
    \item example answer sentences
        \begin{itemize}
            \item Менин жашым жыйырмада. --- I am 20.
            \item Мен жыйырмадамын. --- I am 20.
            \item Ал отузда. --- He / she / it is 30.
            \item Менин апам алтымышта. --- My mother is 60.
            \item Сенин атаң кырк беште. --- Your father is 45.
            \item Сенин атаң канчада? --- How old is your father?
            \item Менин атам алтымыш алтыда. --- My father is 65.
        \end{itemize}
\end{itemize}

\subsection{Asking about family}

\begin{itemize}
    \item in Kyrgyz it is very important to ask about someones family status,
        it is seen as normal and polite -- it can seen rude to foreigners 
    \item examples of such questions are
        \begin{table}[H]
            \center
            \begin{tabular}{| l | l |}\hline
                Kyrgyz              & English       \\\hline\hline
                Үй бүлөң барбы?     & Do you have a family? \\\hline
                Үй бүлөңүз барбы?   & Do you have a family? \\\hline
                Үйлөнгөнсүңбү? / Үйлөнгөнсүзбү? & Are you married? (male speakers)
                \\\hline
                Турмушка чыккансыңбы?   &  Are you married? (female speakers)
                \\
                Турмушка чыккансызбы? &   \\\hline
                Балдарың барбы?     & Do you have children? \\\hline
            \end{tabular}
            \caption{Questions about family status}
        \end{table}
    \item these questions can be answered as follows
        \begin{dialogue}
            \speak{A} Үй бүлөңүз барбы?
            \speak{B} Ооба, үй--бүлөм бар. / Жок, үй--бүлөм жок.
        \end{dialogue}
        \begin{dialogue}
            \speak{A} Үйлөнгөнсүзбү?
            \speak{B} Ооба, үйлөнгөмүн. / Жок, үйлөнө элекмин.
        \end{dialogue}
        \begin{dialogue}
            \speak{A} Турмушка чыккансызбы?
            \speak{B} Ооба, турмушка чыкканмын. / Жок, турмушка чыга элекмин.
        \end{dialogue}
\end{itemize}

\subsubsection{Talking about family members}

    \item Бул менин <status>. Анын аты <name>. Ал <age>. Ал <occupation>.
    \item Бул менин карындашым. Анын аты Эмма. Ал он сегизде. Ал студент.
    \item Бул менин атам. Анын аты Рольф. Ал алтымыш алтыде. Ал профессор.
    \item Бул менин эже. Анын аты Лена. Ал кырк бирде. Ал менеджер.
    \item Бул менин чоң апа. Анын атыи Гретель. Ал токсон төртө. Ал пенсионер.
\end{itemize}
